CREATE TABLE [dbo].[CUSTOMER_DETAILS] (
    [DetailID]   INT IDENTITY (1, 1) NOT NULL,
    [CustomerID] INT NOT NULL,
    [SymptomID]  INT NOT NULL,
    PRIMARY KEY CLUSTERED ([DetailID] ASC),
    CONSTRAINT [FK1CUSTDET] FOREIGN KEY ([CustomerID]) REFERENCES [dbo].[CUSTOMERS] ([CustomerID]),
    CONSTRAINT [FK2CUSTDET] FOREIGN KEY ([SymptomID]) REFERENCES [dbo].[SYMPTOMS] ([SymptomID])
);
