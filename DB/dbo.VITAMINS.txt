CREATE TABLE [dbo].[VITAMINS] (
    [VitaminID]    INT             IDENTITY (1, 1) NOT NULL,
    [VitaminName]  VARCHAR (25)    NOT NULL,
    [Manufacturer] VARCHAR (30)    NOT NULL,
    [Price]        DECIMAL (18, 2) NOT NULL,
    [Amount]       INT             NOT NULL,
    PRIMARY KEY CLUSTERED ([VitaminID] ASC)
);
