CREATE TABLE [dbo].[ORDER_ITEMS] (
    [OrderID]   INT NOT NULL,
    [VitaminID] INT NOT NULL,
    [Quantity]  INT NOT NULL,
    CONSTRAINT [PKORDIT] PRIMARY KEY CLUSTERED ([OrderID] ASC, [VitaminID] ASC),
    CONSTRAINT [FK1ORDIT] FOREIGN KEY ([OrderID]) REFERENCES [dbo].[ORDERS] ([OrderID]),
    CONSTRAINT [FK2ORDIT] FOREIGN KEY ([VitaminID]) REFERENCES [dbo].[VITAMINS] ([VitaminID])
);
